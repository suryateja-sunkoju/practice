user  ->  subscription onetomany mapping of entities.

@Entity
class User{
    @OneToMany
    List<Subscription> subcriptionList;

    @OneToMany
    List<Order> orderList;
}

@Entity
class Meal{
    @Id
    String type;
    String flavour;
    int price;
    @OneToOne
    Discount discount;
}

@Entity
class Subscription{
    @Id
    int durationInMonths;
    int price;

    @ManyToOne
    User user;

    @OneToOne
    Meal meal;
}

enum discountType{
    FIXED,
    PERCENTAGE,
}

@Entity
class Discount{

    @Id
    Long id;
    DiscountType discountType;
    int amount;

    @OneToOne
    Meal meal;
}

@Entity
class Order{
    @Id
    UUID id;
    String mealType;
    int mealPrice;
    int discountAmount;
    int totalPrice;
    int subscriptionPeriod;
    int noOfOneTimeOrders;
    @ManyToOne
    User user;

    @OneToMany
    List<OneTimeOrder> oneTimeOrderList;

}

@Entity
class OneTimeOrder{
    UUID id;
    String mealType;
    @ManyToOne
    Order order;
}
